<script>
  const secret = "showblack_secret";

  function sha256(str) {
    const encoder = new TextEncoder();
    const data = encoder.encode(str);
    return crypto.subtle.digest("SHA-256", data).then(hash => {
      let hex = "";
      const view = new DataView(hash);
      for (let i = 0; i < hash.byteLength; i += 4) {
        hex += ("00000000" + view.getUint32(i).toString(16)).slice(-8);
      }
      return hex;
    });
  }

  function formatDate(date) {
    return date.getUTCFullYear() + "-" +
           String(date.getUTCMonth() + 1).padStart(2, "0") + "-" +
           String(date.getUTCDate()).padStart(2, "0");
  }

  async function generateKey() {
    const today = formatDate(new Date());
    const raw = today + secret;
    const hash = await sha256(raw);
    return hash;
  }

  function copyToClipboard(text) {
    navigator.clipboard.writeText(text).then(() => {
      alert("Key copied to clipboard!");
    });
  }

  async function showHome() {
    const app = document.getElementById("app");
    app.innerHTML = `
      <h1>üîê Among Us Key</h1>
      <p>Click the button below to get your key for today.</p>
      <button id="getKeyBtn">Get Key</button>
    `;

    document.getElementById("getKeyBtn").onclick = () => {
      location.hash = "#/completed-task";
    };
  }

  async function showCompletedTask() {
    const app = document.getElementById("app");
    app.innerHTML = `
      <h1>‚úÖ Task Completed!</h1>
      <p>Generating your key, please wait...</p>
    `;
    // Aguarda 1 segundo antes de redirecionar
    setTimeout(async () => {
      const key = await generateKey();
      location.hash = "#/key/" + key;
    }, 1000);
  }

  async function showKey(key) {
    const app = document.getElementById("app");
    app.innerHTML = `
      <h1>üîê Your Among Us Key for Today</h1>
      <p>Click the key below to copy it.</p>
      <div class="key-box" id="keyBox">${key}</div>
      <button id="backBtn" style="margin-top:20px; font-size:1rem; background:none; border:none; color:#f00; cursor:pointer; text-decoration:underline;">Back to Home</button>
    `;

    document.getElementById("keyBox").onclick = () => copyToClipboard(key);
    document.getElementById("backBtn").onclick = () => {
      location.hash = "#/";
    };
  }

  // Roteador simples usando hash
  async function router() {
    const hash = location.hash || "#/";
    if (hash === "#/" || hash === "") {
      await showHome();
    } else if (hash === "#/completed-task") {
      await showCompletedTask();
    } else if (hash.startsWith("#/key/")) {
      const key = hash.split("/")[2];
      if (key) {
        await showKey(key);
      } else {
        location.hash = "#/";
      }
    } else {
      // Qualquer outro hash volta para home
      location.hash = "#/";
    }
  }

  window.addEventListener("hashchange", router);
  window.addEventListener("load", router);
</script>
